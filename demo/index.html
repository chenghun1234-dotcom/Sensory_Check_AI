// Alexa Skill Lambda (Node.js)
const SensoryCheck = require('./sensory-check-client');

const ProductInfoHandler = {
  canHandle(handlerInput) {
    return Alexa.getRequestType(handlerInput.requestEnvelope) === 'IntentRequest'
      && Alexa.getIntentName(handlerInput.requestEnvelope) === 'ProductInfoIntent';
  },
  async handle(handlerInput) {
    const productUrl = Alexa.getSlotValue(handlerInput.requestEnvelope, 'ProductUrl');
    
    const { voice_summary } = await SensoryCheck.analyze({
      product_url: productUrl,
      focus: ['taste', 'context'],
      language: 'en',
    });
    
    // Deliver natural sensory summary via voice
    return handlerInput.responseBuilder
      .speak(voice_summary.en)
      .getResponse();
  },
};
